348 template <typename Derived1>
349 inline void ALMM::applyPermutation(const Eigen::MatrixBase<Derived1>& M_)
350 {
351  //cast out arguments
352  Eigen::MatrixBase<Derived1>& M = const_cast< Eigen::MatrixBase<Derived1>& >(M_);
353 
354  if(isnull(perm))
355  return;
356 
357  if(perm.rows() != M.rows()){
358  throw CLimixException("ALMM:Permutation vector has incompatible length");
359  }
360  //create temporary copy
361  MatrixXd Mc = M;
362  //apply permutation;
363  for(muint_t i = 0;i < (muint_t)((((Mc.rows()))));++i){
364  M.row(i) = Mc.row(perm(i));
365  }
366 }

371 template <typename Derived, typename OtherDerived>
372 inline void CLMMCore::SelfAdjointEigenSolver(const Eigen::MatrixBase<Derived>& U, const Eigen::MatrixBase<Derived>& S, const Eigen::MatrixBase<OtherDerived>& M)
373 {
374  //1. check size of matrix
375  muint_t dim = M.rows();
376  if (dim==1)
377  {
378  //trivial
379  const_cast<Eigen::MatrixBase<Derived>& >(U) = MatrixXd::Ones(1,1);
380  const_cast<Eigen::MatrixBase<Derived>& >(S) = M;
381  }
382  else if (dim==2)
383  {
384  solver2.computeDirect(M);
385  const_cast<Eigen::MatrixBase<Derived>& >(U) = solver2.eigenvectors();
386  const_cast<Eigen::MatrixBase<Derived>& >(S) = solver2.eigenvalues();
387  }
388  else if (dim==3)
389  {
390  //use eigen direct solver
391  solver3.computeDirect(M);
392  const_cast<Eigen::MatrixBase<Derived>& >(U) = solver3.eigenvectors();
393  const_cast<Eigen::MatrixBase<Derived>& >(S) = solver3.eigenvalues();
394  }
395  else
396  {
397  //use dynamic standard solver
398  Eigen::SelfAdjointEigenSolver<MatrixXd> eigensolver(M);
399  const_cast<Eigen::MatrixBase<Derived>& >(U) = eigensolver.eigenvectors();
400  const_cast<Eigen::MatrixBase<Derived>& >(S) = eigensolver.eigenvalues();
401  }
402 }

405 template <typename Derived1, typename Derived2,typename Derived3,typename Derived4, typename Derived5,typename Derived6,typename Derived7,typename Derived8>
406 inline void CLMMCore::nLLevalEx(const Eigen::MatrixBase<Derived1>& AObeta_, const Eigen::MatrixBase<Derived2>& AObeta_ste_, const Eigen::MatrixBase<Derived3>& AOsigma_, const Eigen::MatrixBase<Derived4>& AOF_tests_,const Eigen::MatrixBase<Derived5>& AOnLL_,const Eigen::MatrixBase<Derived6>& UY, const Eigen::MatrixBase<Derived7>& UX, const Eigen::MatrixBase<Derived8>& S,mfloat_t ldelta,bool calc_ftest,bool calc_ste, bool REML)
407 {
408  //cast out arguments
409  Eigen::MatrixBase<Derived1>& AObeta = const_cast< Eigen::MatrixBase<Derived1>& >(AObeta_);
410  Eigen::MatrixBase<Derived2>& AObeta_ste = const_cast< Eigen::MatrixBase<Derived2>& >(AObeta_ste_);
411  Eigen::MatrixBase<Derived3>& AOsigma = const_cast< Eigen::MatrixBase<Derived3>& >(AOsigma_);
412  Eigen::MatrixBase<Derived4>& AOF_tests = const_cast< Eigen::MatrixBase<Derived4>& >(AOF_tests_);
413  Eigen::MatrixBase<Derived5>& AOnLL = const_cast< Eigen::MatrixBase<Derived5>& >(AOnLL_);
414 
415 
416  //number of samples
417  muint_t n = UX.rows();
418  //number of dimensions for fitting (X)
419  muint_t d = UX.cols();
420  //number of phenotypes to evaluate:
421  muint_t p = UY.cols();
422 
423  //resize output arguments as needed
424  AOnLL.derived().resize(1,p);
425  AObeta.derived().resize(d,p);
426  AOsigma.derived().resize(1,p);
427 
428  if (calc_ftest || calc_ste)
429  AOF_tests.derived().resize(d,p);
430  if (calc_ste)
431  AObeta_ste.derived().resize(d,p);
432 
433 
434  assert(UY.rows() == S.rows());
435  assert(UY.rows() == UX.rows());
436 
437  mfloat_t delta = exp(ldelta);
438  Sdi = S.array() + delta;
439  mfloat_t ldet = 0.0;
440  mfloat_t ldetXX = 0.0;
441  mfloat_t ldetXSX = 0.0;
442  //calc log det and invert at the same time Sdi elementwise
443  for(size_t ind = 0;ind < n;++ind)
444  {
445  //ldet
446  ldet += log(Sdi.data()[ind]);
447  //inverse:
448  Sdi.data()[ind] = 1.0 / (Sdi.data()[ind]);
449  }
450  //check whether ldet is NAN, => set to infinity
451 #ifdef _MSC_VER
452  if (ldet!=ldet)
453  {
454  ldet=std::numeric_limits<mfloat_t>::infinity();
455  }
456 #else
457  if(ldet==NAN)
458  {
459  ldet=INFINITY;
460  }
461 #endif
462 
463  if (calc_ftest || calc_ste)
464  AOF_tests.setConstant(0.0);
465 
466 
467  XSdi = UX.array().transpose();
468  XSdi.array().rowwise() *= Sdi.array().transpose();
469  XSX.noalias() = XSdi * UX;
470  XSY.noalias() = XSdi * UY;
471 
472  //least squares solution of XSX*beta = XSY
473  //Call internal solver which uses fixed solvers for 2d and 3d matrices
474  SelfAdjointEigenSolver(U_X, S_X, XSX);
475  AObeta.noalias() = U_X.transpose() * XSY;
476 
477  //loop over genotype dimensions:
478  for(size_t dim = 0;dim < d;++dim)
479  {
480  if(S_X(dim) > 3E-8)
481  {
482  AObeta.row(dim).array() /= S_X(dim);
483  ldetXSX+=log(S_X(dim));
484  if (calc_ftest || calc_ste)
485  {
486  for(size_t dim2 = 0;dim2 < d;++dim2)
487  AOF_tests.array().row(dim2) += U_X(dim2, dim) * U_X(dim2, dim) / S_X(dim, 0);
488  }
489  }
490  else
491  AObeta.row(dim).setConstant(0.0);
492  }
493 
494  AObeta = U_X * AObeta;
495  res.noalias() = UY - UX * AObeta;
496  //sqared residuals
497  res.array() *= res.array();
498  res.array().colwise() *= Sdi.array();
499 
500  /*
501  std::cout << "UX" << UX << "\n";
502  std::cout << "UY" << UX << "\n";
503  std::cout << "Sdi" << Sdi << "n";
504 
505  std::cout << "AObeta" << AObeta << "\n";
506  std::cout << "res" << res << "n";
507  std::cout << "ldet" << ldet << "\n";
508  */
509 
510  //compute REML/ML specific terms
511  if (REML){
512  AOsigma = res.colwise().sum() / (mfloat_t)(n-d);
513 
514  MatrixXd XX = UX.transpose() * UX;
515  Eigen::SelfAdjointEigenSolver<MatrixXd> eigensolver(XX);
516  //MatrixXd U_XX = eigensolver.eigenvectors();
517  MatrixXd S_XX = eigensolver.eigenvalues();
518  for(size_t dim = 0;dim < d;++dim){
519  if(S_XX(dim, 0) > 3E-8){
520  ldetXX += log(S_XX(dim, 0));
521  }
522  }
523 
524  }else{
525  AOsigma = res.colwise().sum() / (mfloat_t)n;
526  }
527 
528  //compute standard errors
529  if(calc_ste)
530  {
531  AObeta_ste = AOF_tests.cwiseInverse();
532  AObeta_ste.array().rowwise() *= AOsigma.array().row(0);
533  //Christoph: I think this is a bug:
534  //ste should not be divided by additional n, That factor is already considered in sigma
535  //if (REML){
536  // AObeta_ste/=n-d;
537  //}else{
538  // AObeta_ste/=n;
539  //}
540  AObeta_ste =AObeta_ste.cwiseSqrt();
541  }
542  //compute the F-statistics
543  if(calc_ftest)
544  {
545  AOF_tests.array() = AObeta.array() * AObeta.array() / AOF_tests.array();
546  AOF_tests.array().rowwise() /= AOsigma.array().row(0);
547  }
548 
549 
550  //WARNING: elementwise log of sigg2
551  logInplace(AOsigma);
552  //calc likelihood:
553  if (REML){
554  AOnLL = 0.5*(n-d)*AOsigma;
555  AOnLL.array() += 0.5*((n-d)*L2pi+ldet +ldetXX + ldetXSX+ (n-d));
556  }else{
557  AOnLL = 0.5*n*AOsigma;
558  AOnLL.array() += 0.5*(n*L2pi+ldet + n);
559  }
560 }

566 inline void CLMMCore::optdeltaEx(const Eigen::MatrixBase<Derived1> & AO_delta_,const Eigen::MatrixBase<Derived2> & AO_NLL_, const Eigen::MatrixBase<Derived3>& UY,const Eigen::MatrixBase<Derived4>& UX,const Eigen::MatrixBase<Derived5>& S,muint_t numintervals,mfloat_t ldeltamin,mfloat_t ldeltamax,bool REML)

568 
569  //cast out arguments
570  Eigen::MatrixBase<Derived1>& AO_delta = const_cast< Eigen::MatrixBase<Derived1>& >(AO_delta_);
571  Eigen::MatrixBase<Derived1>& AO_NLL = const_cast< Eigen::MatrixBase<Derived1>& >(AO_NLL_);
572 
573 
574  //number of phenotypes
575  muint_t Np = UY.cols();
576 
577  //grid variable with the current likelihood evaluations
578  AO_delta.derived().resize(Np,1);
579  AO_NLL.derived().resize(Np,1);
580 
581 
582  nllgrid.resize(numintervals,Np);
583  ldeltagrid.resize(numintervals);
584  //current delta
585  mfloat_t ldelta = ldeltamin;
586  mfloat_t ldeltaD = (ldeltamax - ldeltamin);
587  ldeltaD /= ((mfloat_t)(numintervals) - 1.0);
588 
589  MatrixXd f_tests_,AObeta_,AObeta_ste_,AOsigma_;
590 
591  //forall elements in grid do:
592  for(muint_t ii = 0;ii < (numintervals);++ii)
593  {
594  ldeltagrid(ii) = ldelta;
595  //get nll for all phenotypes jointly (using current ldelta)
596  this->nLLevalEx(f_tests_,AObeta_,AObeta_ste_,AOsigma_, nllgrid.block(ii,0,1,Np),UY, UX, S,ldelta,false,REML);
597  //move on delta
598  ldelta += ldeltaD;
599  }
600  //std::cout << nllgrid << "\n";

603  for (muint_t ip=0;ip<Np;++ip)

606  AO_delta(ip) = ldeltagrid(min_index);
607  /*
608  std::cout << nllgrid.col(ip) << "\n";
609  std::cout << "\n\n";
610  std::cout << "min index:" << min_index << "\n";
611  std::cout << ldeltagrid << "\n";
612  std::cout << "delta:"<< AO_delta(ip) << "\n";
613  */
614  }
615  //std::cout << AO_NLL << "\n";
616  //std::cout << AO_delta << "\n";
617

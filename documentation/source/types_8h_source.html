231 /* CParamObject:
232  * - provides basic handling for synchronization of precalculated data
233  * syncParents: boolen sync states the computation of this object depends on
234  * syncChildren: children object whose calculations depend on (this)
235  *
236  * addSyncParent / addSyncChild: add a sync element
237  * isInSync(): checks that all parent sync objects are true (i.e. does the state of hte current object need to updated?)
238  * makeSync(): sets all parent sync true (i.e. set the current object to have an up2date state)
239  */
240 class CParamObject
241 {
242 protected:
243  Pbool sync;
244  //paranets: objects that require sync above the hierarchy of this one
245  PboolVec syncParents;
246  //paranets: objects that require sync below the hierarchy of this one
247  PboolVec syncChildren;
248  //propagateSync: set all children sync states to false to force them update computations
249 public:
250 
251  CParamObject()
252  {
253  sync = Pbool(new bool);
254  //add standard sync paranet which is the own sync variable
255  addSyncParent(sync);
256  }
